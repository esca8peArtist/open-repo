# measurement_env_template.txt
# Environment variable template for Phase 1 measurement automation
# Copy this file to measurement_env.sh, fill in the values, and source it:
#   source /path/to/measurement_env.sh
#
# NEVER commit the filled-in version to git. Add measurement_env.sh to .gitignore.

# ---------------------------------------------------------------------------
# Kit (ConvertKit) API credentials
# ---------------------------------------------------------------------------
# Get from: app.kit.com -> Settings -> Developer -> API Secret (v3)
# The v3 API Secret is sufficient for read-only broadcast stats (no OAuth needed).
# Rate limit: 120 requests/minute on free tier; 300/minute on Creator tier.
export KIT_API_SECRET="YOUR_KIT_API_SECRET_HERE"

# ---------------------------------------------------------------------------
# Google Sheets
# ---------------------------------------------------------------------------
# SPREADSHEET_ID: the long string in the Google Sheets URL between /d/ and /edit
#   e.g. https://docs.google.com/spreadsheets/d/1BxiMVs0XRA5nFMdKvBdBZjgmUUqptlbs74OgVE2upms/edit
#   => SPREADSHEET_ID = "1BxiMVs0XRA5nFMdKvBdBZjgmUUqptlbs74OgVE2upms"
export SPREADSHEET_ID="YOUR_GOOGLE_SHEETS_ID_HERE"

# SHEETS_CREDS_JSON: full path to the Google Service Account JSON key file.
# How to generate:
#   1. Go to console.cloud.google.com -> IAM & Admin -> Service Accounts
#   2. Create a service account -> Add key -> JSON
#   3. Download the JSON file to this directory
#   4. Share the Google Sheet with the service account email (Editor role)
export SHEETS_CREDS_JSON="/home/awank/dev/SuperClaude_Framework/projects/cybersecurity-hardening/config/service_account.json"

# ---------------------------------------------------------------------------
# Discord webhook
# ---------------------------------------------------------------------------
# How to create:
#   1. Open Discord -> Server Settings -> Integrations -> Webhooks
#   2. Create webhook, choose the channel, copy the webhook URL
#   3. Test: curl -X POST -H 'Content-Type: application/json' \
#            -d '{"content":"test"}' YOUR_WEBHOOK_URL
export DISCORD_WEBHOOK_URL="https://discord.com/api/webhooks/YOUR_WEBHOOK_ID/YOUR_WEBHOOK_TOKEN"
